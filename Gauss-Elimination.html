Code:


        
#include <stdio.h>

    int main() {
        int n, i, j, k;
        printf("Enter the number of equations: ");
        scanf("%d", &n);
    
        double a[n][n + 1];  // Augmented matrix
        double x[n];         // Solution array
    
        // Input augmented matrix
        printf("Enter the elements of the augmented matrix row by row:\n");
        for (i = 0; i < n; i++) {
            for (j = 0; j < n + 1; j++) {
                scanf("%lf", &a[i][j]);
            }
        }
    
        // Applying Gaussian Elimination
        for (i = 0; i < n - 1; i++) {
            for (k = i + 1; k < n; k++) {
                double factor = a[k][i] / a[i][i];
                for (j = i; j < n + 1; j++) {
                    a[k][j] -= factor * a[i][j];
                }
            }
        }
    
        // Back substitution
        for (i = n - 1; i >= 0; i--) {
            x[i] = a[i][n];
            for (j = i + 1; j < n; j++) {
                x[i] -= a[i][j] * x[j];
            }
            x[i] /= a[i][i];
        }
    
        // Displaying the solution
        printf("Solution:\n");
        for (i = 0; i < n; i++) {
            printf("x[%d] = %.6f\n", i + 1, x[i]);
        }
    
        return 0;
    }
