Bisection Mtethod in C

Code:


        
#include <stdio.h>

    double f(double x) {
        return x * x; // Example function: f(x) = x^2
    }
    
    int main() {
        int n;
        double a, b, h, integral = 0.0;
    
        printf("Enter the lower limit of integration (a): ");
        scanf("%lf", &a);
        
        printf("Enter the upper limit of integration (b): ");
        scanf("%lf", &b);
        
        printf("Enter the number of subintervals (n): ");
        scanf("%d", &n);
    
        h = (b - a) / n;
    
        integral += 0.5 * (f(a) + f(b));
    
        for (int i = 1; i < n; i++) {
            double x = a + i * h;
            integral += f(x);
        }
    
        integral *= h;
    
        printf("The approximate value of the integral is: %.6lf\n", integral);
        return 0;
    }
