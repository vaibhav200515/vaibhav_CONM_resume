floating point representation binary to decimal in C

Code:

#include <stdio.h>
//floating point representation binary to decimal

void main()
{
	int bm,be;
	int nm=0,rem,pow=1;
	float ne=0.0,powe=2.0,num;
	int temp=10000,i;
	
	printf("Enter binary of mantissa : ");
	scanf("%d",&bm);
	printf("Enter binary of exponent (5 digits) : ");
	scanf("%d",&be);
	printf("Entered binary number : %d.%d",bm,be);
	
	while(bm>0)
	{
		rem=bm%10;
		bm=bm/10;
		nm=nm+rem*pow;
		pow=pow*2;
	}
	
	for(i=0;i<5;i++)
	{
		rem=be/temp;
		be=be%temp;
		temp=temp/10;
		ne=ne+rem/powe;
		powe=powe*2.0;
	}
	
	num=nm+ne;
	printf("\nDecimal of entered number : %f",num);
}
