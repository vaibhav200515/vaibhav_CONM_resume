Bisection Mtethod in C

Code:


        
#include <stdio.h>
#include <math.h>
  
    int main() {
        int n, i, j, iterations;
        double tolerance;
    
        printf("Enter the number of equations: ");
        scanf("%d", &n);
    
        double a[n][n], b[n], x[n], newX[n];
    
        // Input coefficients of the matrix
        printf("Enter the coefficients of the augmented matrix:\n");
        for (i = 0; i < n; i++) {
            for (j = 0; j < n; j++) {
                printf("a[%d][%d]: ", i, j);
                scanf("%lf", &a[i][j]);
            }
            printf("b[%d]: ", i);
            scanf("%lf", &b[i]);
            x[i] = 0;  // Initial guess for solution
        }
    
        printf("Enter the tolerance (e.g., 1e-6): ");
        scanf("%lf", &tolerance);
        printf("Enter maximum number of iterations: ");
        scanf("%d", &iterations);
    
        int k;
        for (k = 0; k < iterations; k++) {
            int converged = 1;  // Assume solution is converging
            for (i = 0; i < n; i++) {
                newX[i] = b[i];
                for (j = 0; j < n; j++) {
                    if (i != j) {
                        newX[i] -= a[i][j] * x[j];
                    }
                }
                newX[i] /= a[i][i];
    
                // Check convergence
                if (fabs(newX[i] - x[i]) > tolerance) {
                    converged = 0;
                }
    
                // Update the current value
                x[i] = newX[i];
            }
    
            // If solution has converged, break the loop
            if (converged) {
                break;
            }
        }
    
        // Display the solution
        printf("Solution after %d iterations:\n", k + 1);
        for (i = 0; i < n; i++) {
            printf("x[%d] = %.6f\n", i + 1, x[i]);
        }
    
        return 0;
    }
