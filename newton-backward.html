Bisection Mtethod in C

Code:


        
#include <stdio.h>

    int main() {
        int n;
        double x[10], y[10], diff[10][10], value, h, result, term;
    
        printf("Enter the number of data points (n): ");
        scanf("%d", &n);
    
        printf("Enter the x values:\n");
        for (int i = 0; i < n; i++) {
            scanf("%lf", &x[i]);
        }
        
        printf("Enter the y values:\n");
        for (int i = 0; i < n; i++) {
            scanf("%lf", &y[i]);
        }
    
        for (int i = 0; i < n; i++) {
            diff[i][0] = y[i];
        }
    
        for (int j = 1; j < n; j++) {
            for (int i = n - 1; i >= j; i--) {
                diff[i][j] = diff[i][j - 1] - diff[i - 1][j - 1];
            }
        }
    
        printf("Enter the value of x for interpolation: ");
        scanf("%lf", &value);
    
        h = x[1] - x[0];
        result = y[n - 1];
        term = 1.0;
    
        for (int i = 1; i < n; i++) {
            term *= (value - x[n - i]) / (h * i);
            result += term * diff[n - 1][i];
        }
    
        printf("The interpolated value at x = %.2lf is: %.6lf\n", value, result);
        return 0;
    }
