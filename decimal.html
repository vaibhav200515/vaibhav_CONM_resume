Decimal to Binary Conversion in C

Code:

#include <stdio.h>

void main() {
    int num1, num2, t, bin = 0, pow = 1;
    
    printf("Enter any number: ");
    scanf("%d", &num1);
    
    num2 = num1;
    
    while (num2 > 0) {
        t = num2 % 2;      
        num2 = num2 / 2;   
        bin = bin + t * pow;
        pow = pow * 10;    
    }
    
    printf("Binary of %d is: %d", num1, bin);
}

Output:

Enter any number: 13
Binary of 13 is: 1101
