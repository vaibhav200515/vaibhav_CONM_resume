#include <stdio.h>
#include <math.h>
  
    int main() {
        int degree, i, iterations;
        double tolerance, x0, x1;
    
        printf("Enter the degree of the polynomial: ");
        scanf("%d", °ree);
    
        double coeff[degree + 1];
        printf("Enter the coefficients of the polynomial from highest degree to constant term:\n");
        for (i = 0; i <= degree; i++) {
            printf("Coefficient of x^%d: ", degree - i);
            scanf("%lf", &coeff[i]);
        }
    
        printf("Enter the initial guess: ");
        scanf("%lf", &x0);
        printf("Enter the tolerance (e.g., 1e-6): ");
        scanf("%lf", &tolerance);
        printf("Enter the maximum number of iterations: ");
        scanf("%d", &iterations);
    
        int k;
        for (k = 0; k < iterations; k++) {
            double b[degree + 1], c[degree + 1];
    
            // Synthetic division for b
            b[0] = coeff[0];
            for (i = 1; i <= degree; i++) {
                b[i] = coeff[i] + b[i - 1] * x0;
            }
    
            // Synthetic division for c (derivative)
            c[0] = b[0];
            for (i = 1; i < degree; i++) {
                c[i] = b[i] + c[i - 1] * x0;
            }
    
            // Update the root approximation
            x1 = x0 - b[degree] / c[degree - 1];
    
            // Check for convergence
            if (fabs(x1 - x0) < tolerance) {
                printf("Root found: %.6f\n", x1);
                return 0;
            }
    
            x0 = x1;
        }
    
        printf("Root not found within %d iterations\n", iterations);
        return 0;
    }
