Code:

#include <stdio.h>
#include <math.h>

float convert(float x1, float x2)
{
	float sum=x1-x2;
	if(sum<0)
	{
		return -sum;
	}
	else
	{
		return sum;
	}
}

void main()
{	
	int x,i,j,c=0;
	int coff[10];
	int sum;
	float a,b,ans;
	float y1,y2;
	
	printf("Enter largest exponent : ");
	scanf("%d",&x);
	
	//getting cofficients
	for(i=x;i>=0;i--)
	{
		printf("Enter cofficient of %d : ",i);
		scanf("%d",&coff[i]);
	}
	
	//print equation
	printf("\nEntered equation : ");
	for(i=x;i>=0;i--)
	{
		if(i>0)
		{
			printf("(%dx^%d) + ",coff[i],i);
		}
		else
		{
			printf("(%d) = 0",coff[i]);
		}
	}
	
	//solving equation
	for(j=-5;j<=5;j++)
	{
		sum=0;
		for(i=x;i>=0;i--)
		{
			sum=sum+pow(j,i)*coff[i];
		}
		printf("\nx = %d \t f(%d) = %d",j,j,sum);
	}
	
	printf("\nEnter value of a (f(x)<0) : ");
	scanf("%f",&a);
	printf("Enter value of b (f(x)>0) : ");
	scanf("%f",&b);
	printf("\na=%.2f\tb=%.2f",a,b);
	
	do
	{
		y1=(a+b)/2;
		c++;
		ans=0.0;
		printf("\nIteration %d ",c);
		for(i=x;i>=0;i--)
		{
			ans=ans+pow(y1,i)*coff[i];
		}
		if(ans>0)
		{
			b=y1;
		}
		else if(ans<0)
		{
			a=y1;
		}
		y2=(a+b)/2;
		printf("\nx%d=%.4f \t f(%.4f)=%.4f",c,y1,y1,ans);
		printf("\na=%.2f\tb=%.2f \t x%d=%.4f",a,b,c+1,y2);
	}while(convert(y1,y2)>0.002);
	
	printf("\n\nRoot of the equation : %.4f",y2);
}
