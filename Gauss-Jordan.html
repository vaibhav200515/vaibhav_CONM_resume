Bisection Mtethod in C

Code:


        
#include <stdio.h>

    int main() {
        int n, i, j, k;
        printf("Enter the number of equations: ");
        scanf("%d", &n);
    
        double a[n][n + 1];  // Augmented matrix
    
        // Input augmented matrix
        printf("Enter the elements of the augmented matrix row by row:\n");
        for (i = 0; i < n; i++) {
            for (j = 0; j < n + 1; j++) {
                scanf("%lf", &a[i][j]);
            }
        }
    
        // Applying Gauss-Jordan Elimination
        for (i = 0; i < n; i++) {
            // Make the diagonal element 1 and normalize the row
            double pivot = a[i][i];
            for (j = 0; j < n + 1; j++) {
                a[i][j] /= pivot;
            }
    
            // Make all elements in the current column (except the pivot) zero
            for (k = 0; k < n; k++) {
                if (k != i) {
                    double factor = a[k][i];
                    for (j = 0; j < n + 1; j++) {
                        a[k][j] -= factor * a[i][j];
                    }
                }
            }
        }
    
        // Displaying the solution
        printf("Solution:\n");
        for (i = 0; i < n; i++) {
            printf("x[%d] = %.6f\n", i + 1, a[i][n]);
        }
    
        return 0;
    }
