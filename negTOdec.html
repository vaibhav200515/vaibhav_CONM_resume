Negative Binary to Decimal Conversion in C

Code:

#include <stdio.h>

void main()
{
    int bin1, bin2, c;
    int bin[8] = {0}, i, pow = 10000000, t;
    int num = 0;
    
    printf("Enter binary of negative number (8 bits): ");
    scanf("%d", &bin1);
    
    bin2 = bin1;
    i = 0;
    while (bin2 > 0)
    {
        t = bin2 / pow;
        bin2 = bin2 % pow;
        pow = pow / 10;
        bin[i] = t;
        i++;
    }
    
    for (i = 0; i < 8; i++)
    {
        if (bin[i] == 1)
        {
            bin[i] = 0;
        }
        else
        {
            bin[i] = 1;
        }
    }
    
    printf("\n1's complement: ");
    for (i = 0; i < 8; i++)
    {
        printf("%d", bin[i]);
    }
    
    c = 1;
    i = 7;
    while (c != 0)
    {
        if (bin[i] == 1)
        {
            bin[i] = 0;
            i--;
        }
        else
        {
            bin[i] = 1;
            c = 0;
            break;
        }
    }
    
    printf("\n2's complement: ");
    for (i = 0; i < 8; i++)
    {
        printf("%d", bin[i]);
    }
    
    i = 7;
    pow = 1;
    while (i >= 0)
    {
        num = num + bin[i] * pow;
        pow *= 2;
        i--;
    }
    printf("\nDecimal of entered binary: %d", -num);
}
