Bisection Mtethod in C

Code:


        
#include <stdio.h>

    int main() {
        int n;
        double x[10], y[10], value, result;
    
        printf("Enter the number of data points (n): ");
        scanf("%d", &n);
    
        printf("Enter the x values:\n");
        for (int i = 0; i < n; i++) {
            scanf("%lf", &x[i]);
        }
    
        printf("Enter the y values:\n");
        for (int i = 0; i < n; i++) {
            scanf("%lf", &y[i]);
        }
    
        printf("Enter the value of y for interpolation: ");
        scanf("%lf", &value);
    
        result = 0.0;
    
        for (int i = 0; i < n; i++) {
            double term = x[i];
            for (int j = 0; j < n; j++) {
                if (j != i) {
                    term *= (value - y[j]) / (y[i] - y[j]);
                }
            }
            result += term;
        }
    
        printf("The interpolated value at y = %.2lf is: %.6lf\n", value, result);
        return 0;
    }
