Bisection Mtethod in C

Code:


        
#include <stdio.h>

    int main() {
        double x0, y0, x, h, y, k1, k2, k3, k4;
        int n;
    
        printf("Enter the initial value of x (x0): ");
        scanf("%lf", &x0);
        
        printf("Enter the initial value of y (y0): ");
        scanf("%lf", &y0);
        
        printf("Enter the step size (h): ");
        scanf("%lf", &h);
        
        printf("Enter the value of x to which you want to find y: ");
        scanf("%lf", &x);
        
        n = (int)((x - x0) / h);
        y = y0;
    
        for (int i = 0; i < n; i++) {
            k1 = 2 * y; // Replace (2 * y) with your specific f(x, y)
            k2 = 2 * (y + (h / 2) * k1);
            k3 = 2 * (y + (h / 2) * k2);
            k4 = 2 * (y + h * k3);
            
            y = y + (h / 6) * (k1 + 2 * k2 + 2 * k3 + k4); // Update y
            x0 = x0 + h; // Move to the next x value
        }
    
        printf("The approximate value of y at x = %.2lf is: %.6lf\n", x, y);
        return 0;
    }
