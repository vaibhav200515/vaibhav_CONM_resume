Binary to Decimal Conversion in C

Code:

#include <stdio.h>

void main() {
    int bin1, bin2, t, num = 0, pow = 1;

    printf("Enter any binary number: ");
    scanf("%d", &bin1);

    bin2 = bin1;

    while (bin2 > 0) {
        t = bin2 % 10;
        bin2 = bin2 / 10;
        num = num + t * pow;
        pow = pow * 2;
    }

    printf("Decimal of %d is: %d", bin1, num);
}

Output:

Enter any binary number: 1101
Decimal of 1101 is: 13
