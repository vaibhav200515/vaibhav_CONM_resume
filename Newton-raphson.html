Bisection Mtethod in C

Code:


        
#include <stdio.h>
#include <math.h>
  
    int main() {
        double x0, tolerance;
        int maxIterations;
    
        // Take user inputs
        printf("Enter initial guess: ");
        scanf("%lf", &x0);
        printf("Enter tolerance (e.g., 1e-6): ");
        scanf("%lf", &tolerance);
        printf("Enter maximum iterations: ");
        scanf("%d", &maxIterations);
    
        double x1;
        int i;
    
        for (i = 0; i < maxIterations; i++) {
            double fx = x0 * x0 * x0 - x0 - 1;           // f(x) = x^3 - x - 1
            double dfx = 3 * x0 * x0 - 1;                 // f'(x) = 3x^2 - 1
    
            x1 = x0 - fx / dfx;                           // Newton-Raphson formula
    
            if (fabs(x1 - x0) < tolerance) {              // Check for convergence
                printf("Root found: %.6f\n", x1);
                return 0;
            }
    
            x0 = x1;                                      // Update x0
        }
    
        printf("Root not found within %d iterations\n", maxIterations);
        return 0;
    }
