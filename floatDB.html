Decimal to Binary Conversion (Floating Point) in C

Code:

#include <stdio.h>

void main()
{
    int num;
    int rem, pow = 1;
    long int binm = 0;
    int i, exp, man;
    int bine[5];
    
    printf("Enter any number (4 digits Mantissa 2 digits Exponent): ");
    scanf("%d", &num);
    exp = num % 100;
    man = num / 100;
    printf("Entered number: %d.%d", man, exp);
    
    while (man > 0)
    {
        rem = man % 2;
        man = man / 2;
        binm = binm + rem * pow;
        pow = pow * 10;
    }
    
    for (i = 0; i < 5; i++)
    {
        exp = 2 * exp;
        rem = exp / 100;
        exp = exp % 100;
        bine[i] = rem;
    }
    
    printf("\nBinary of entered decimal: %d.", binm);
    for (i = 0; i < 5; i++)
    {
        printf("%d", bine[i]);
    }
}

Output:
