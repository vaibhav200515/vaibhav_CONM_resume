Code:


        
#include <stdio.h>
#include <math.h>
   
    // Define the function to integrate, e.g., f(x) = x^3
    double f(double x) {
        return x * x * x;  // Example function x^3
    }
    
    int main() {
        double a, b;
        int n, i;
    
        // Input the limits and number of intervals
        printf("Enter the lower limit of integration (a): ");
        scanf("%lf", &a);
        printf("Enter the upper limit of integration (b): ");
        scanf("%lf", &b);
        printf("Enter the number of intervals (n, must be a multiple of 3): ");
        scanf("%d", &n);
    
        // Check if n is a multiple of 3
        if (n % 3 != 0) {
            printf("Number of intervals must be a multiple of 3.\n");
            return 1;
        }
    
        double h = (b - a) / n;
        double sum = f(a) + f(b);
    
        // Calculate the sum for the intermediate terms
        for (i = 1; i < n; i++) {
            double x = a + i * h;
            if (i % 3 == 0) {
                sum += 2 * f(x);  // Terms where index is a multiple of 3
            } else {
                sum += 3 * f(x);  // Other terms
            }
        }
    
        // Apply Simpson's 3/8 Rule
        double integral = (3 * h / 8) * sum;
    
        // Output the result
        printf("The approximate integral is: %.6f\n", integral);
    
        return 0;
    }
