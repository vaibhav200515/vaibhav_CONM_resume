Negative Decimal to Binary Conversion in C

Code:

#include <stdio.h>

void main()
{
    int num1, num2;
    int t, bin[8] = {0}, i = 7, c;
    
    printf("Enter a negative number: ");
    scanf("%d", &num1);
    printf("Entered negative number: %d", num1);
    
    num2 = -num1;
    
    while (num2 > 0)
    {
        t = num2 % 2;
        num2 = num2 / 2;
        bin[i] = t;
        i--;
    }
    
    printf("\nBinary of positive number: ");
    for (i = 0; i < 8; i++)
    {
        printf("%d", bin[i]);
    }
    
    for (i = 0; i < 8; i++)
    {
        if (bin[i] == 1)
        {
            bin[i] = 0;
        }
        else
        {
            bin[i] = 1;
        }
    }
    
    printf("\n1's complement: ");
    for (i = 0; i < 8; i++)
    {
        printf("%d", bin[i]);
    }

    c = 1;
    i = 7;
    while (c != 0)
    {
        if (bin[i] == 1)
        {
            bin[i] = 0;
            i--;
        }
        else
        {
            bin[i] = 1;
            c = 0;
            break;
        }
    }
    
    printf("\n2's complement: ");
    for (i = 0; i < 8; i++)
    {
        printf("%d", bin[i]);
    }
}
